Usage 
import 'package:flutter/material.dart';
import 'package:larkfinserv_flutter_sdk/larkfinserv_flutter_sdk.dart';

class MyApp extends StatefulWidget {
  @override
  _MyAppState createState() => _MyAppState();
}

class _MyAppState extends State {
  late LarkFinServSDK _sdk;
  bool _isInitialized = false;

  @override
  void initState() {
    super.initState();
    _sdk = LarkFinServSDK();
    _initializeSDK();
  }

  void _initializeSDK() async {
    try {
      await _sdk.initialize(PartnerConfig(
        apiKey: 'your-api-key',
        apiSecret: 'your-api-secret',
        partnerId: 'your-partner-id',
        phoneNumber: '+919876543210',
      ));
      setState(() {
        _isInitialized = true;
      });
    } catch (e) {
      print('SDK initialization failed: $e');
    }
  }

  void _startEligibilityCheck() {
    if (_isInitialized) {
      _sdk.openEligibilityCheck(SDKMode.popup);
    }
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(title: Text('Lark SDK Demo')),
        body: Center(
          child: _isInitialized
              ? EligibilityCheckWidget(
                  sdk: _sdk,
                  onEvent: (event) {
                    print('SDK Event: ${event.type}');
                  },
                )
              : CircularProgressIndicator(),
        ),
      ),
    );
  }

  @override
  void dispose() {
    _sdk.dispose();
    super.dispose();
  }
}

Integration

Complete integration example with React hooks and state management

App.tsx 
import React, { useState, useEffect } from 'react';
import LoanEligibilitySDK from 'loan-eligibility-sdk';
import type { SDKEvent } from 'loan-eligibility-sdk/src/types';

function App() {
  const [sdk, setSdk] = useState<LoanEligibilitySDK | null>(null);
  const [status, setStatus] = useState<'idle' | 'loading' | 'success' | 'error'>('idle');
  const [phoneNumber, setPhoneNumber] = useState('');

  useEffect(() => {
    if (phoneNumber.length !== 10) return;

    const sdkInstance = new LoanEligibilitySDK({
      apiKey: process.env.REACT_APP_SDK_KEY!,
      apiSecret: process.env.REACT_APP_SDK_SECRET!,
      phoneNumber: `+91${phoneNumber}`,
    });

    // Event handlers
    sdkInstance.on('INITIATED', () => setStatus('loading'));
    sdkInstance.on('READY', () => setStatus('success'));
    sdkInstance.on('ELIGIBILITY_RESULT', (event: SDKEvent) => {
      console.log('Result:', event.data);
      // Handle eligibility result
    });
    sdkInstance.on('ERROR', (event: SDKEvent) => {
      console.error('Error:', event.data);
      setStatus('error');
    });

    setSdk(sdkInstance);
  }, [phoneNumber]);

  const handleStartCheck = async () => {
    if (!sdk) return;

    try {
      await sdk.initialize({
        partnerId: process.env.REACT_APP_PARTNER_ID!,
        apiKey: process.env.REACT_APP_SDK_KEY!,
        apiSecret: process.env.REACT_APP_SDK_SECRET!,
      });

      sdk.openEligibilityCheck('inline');
    } catch (error) {
      console.error('Failed to start check:', error);
      setStatus('error');
    }
  };

  return (
    <div>
      <input
        type="tel"
        value={phoneNumber}
        onChange={(e) => setPhoneNumber(e.target.value.replace(/\D/g, ''))}
        placeholder="Enter 10-digit mobile number"
        maxLength={10}
      />
      
      <button
        onClick={handleStartCheck}
        disabled={phoneNumber.length !== 10 || status === 'loading'}
      >
        {status === 'loading' ? 'Loading...' : 'Start Eligibility Check'}
      </button>
      
      <div id="loan-eligibility-sdk"></div>
    </div>
  );
}
Advanced Integration with State Persistence 
import React, { useState, useEffect, useRef } from 'react';
import LoanEligibilitySDK from 'loan-eligibility-sdk';
import type { SDKEvent } from 'loan-eligibility-sdk/src/types';

type SDKStatus = 'idle' | 'loading' | 'success' | 'error' | 'completed';

function App() {
  const [sdk, setSdk] = useState<LoanEligibilitySDK | null>(null);
  const restorationAttempted = useRef(false);
  
  // State persistence
  const [status, setStatus] = useState<SDKStatus>(() => {
    const saved = localStorage.getItem('loanSDK_status');
    return saved ? (saved as SDKStatus) : 'idle';
  });
  
  const [showSDK, setShowSDK] = useState(() => {
    const saved = localStorage.getItem('loanSDK_showSDK');
    return saved ? JSON.parse(saved) : false;
  });
  
  const [phoneNumber, setPhoneNumber] = useState(() => {
    const saved = localStorage.getItem('loanSDK_phoneNumber');
    return saved || '';
  });

  // Persist state
  useEffect(() => {
    localStorage.setItem('loanSDK_status', status);
  }, [status]);

  useEffect(() => {
    localStorage.setItem('loanSDK_showSDK', JSON.stringify(showSDK));
  }, [showSDK]);

  useEffect(() => {
    localStorage.setItem('loanSDK_phoneNumber', phoneNumber);
  }, [phoneNumber]);

  useEffect(() => {
    if (phoneNumber.length !== 10) return;

    const sdkInstance = new LoanEligibilitySDK({
      apiKey: process.env.REACT_APP_SDK_KEY!,
      apiSecret: process.env.REACT_APP_SDK_SECRET!,
      phoneNumber: `+91${phoneNumber}`,
    });

    // Event handlers
    const handlers = {
      initiated: () => {
        setStatus('loading');
        setShowSDK(true);
      },
      ready: () => setStatus('success'),
      result: (event: SDKEvent) => {
        setStatus('completed');
        console.log('Eligibility result:', event.data);
        
        // Auto-redirect after delay
        setTimeout(() => {
          clearSDKState();
          window.location.href = '/dashboard';
        }, 5000);
      },
      error: (event: SDKEvent) => {
        setStatus('error');
        console.error('SDK Error:', event.data);
      },
      close: () => {
        if (status !== 'completed') {
          setStatus('idle');
          setShowSDK(false);
        }
      }
    };

    sdkInstance.on('INITIATED', handlers.initiated);
    sdkInstance.on('READY', handlers.ready);
    sdkInstance.on('ELIGIBILITY_RESULT', handlers.result);
    sdkInstance.on('ERROR', handlers.error);
    sdkInstance.on('CLOSE_FRAME', handlers.close);

    setSdk(sdkInstance);

    // Restore session on page refresh
    if (showSDK && !restorationAttempted.current) {
      restorationAttempted.current = true;
      setTimeout(async () => {
        try {
          await sdkInstance.initialize({
            partnerId: process.env.REACT_APP_PARTNER_ID!,
            apiKey: process.env.REACT_APP_SDK_KEY!,
            apiSecret: process.env.REACT_APP_SDK_SECRET!,
          });
          sdkInstance.openEligibilityCheck('inline');
        } catch (error) {
          console.error('Restore failed:', error);
          setStatus('error');
        }
      }, 200);
    }

    return () => {
      sdkInstance.off('INITIATED', handlers.initiated);
      sdkInstance.off('READY', handlers.ready);
      sdkInstance.off('ELIGIBILITY_RESULT', handlers.result);
      sdkInstance.off('ERROR', handlers.error);
      sdkInstance.off('CLOSE_FRAME', handlers.close);
    };
  }, [phoneNumber]);

  const clearSDKState = () => {
    localStorage.removeItem('loanSDK_status');
    localStorage.removeItem('loanSDK_showSDK');
    localStorage.removeItem('loanSDK_phoneNumber');
  };

  const handleStartCheck = async () => {
    if (!sdk) return;

    try {
      await sdk.initialize({
        partnerId: process.env.REACT_APP_PARTNER_ID!,
        apiKey: process.env.REACT_APP_SDK_KEY!,
        apiSecret: process.env.REACT_APP_SDK_SECRET!,
      });

      sdk.openEligibilityCheck('inline');
    } catch (error) {
      console.error('Failed to start check:', error);
      setStatus('error');
    }
  };

  return (
    <div>
      {!showSDK ? (
        <div>
          <input
            type="tel"
            value={phoneNumber}
            onChange={(e) => setPhoneNumber(e.target.value.replace(/\D/g, ''))}
            placeholder="Enter 10-digit mobile number"
            maxLength={10}
          />
          
          <button
            onClick={handleStartCheck}
            disabled={phoneNumber.length !== 10 || status === 'loading'}
          >
            {status === 'loading' ? 'Loading...' : 'Start Eligibility Check'}
          </button>
        </div>
      ) : (
        <div>
          <div className="progress-bar">
            <div className={`progress ${status === 'completed' ? 'completed' : ''}`}></div>
          </div>
          <div id="loan-eligibility-sdk"></div>
        </div>
      )}
    </div>
  );
}
Custom Hook: useLoanSDK 
import { useState, useEffect, useRef } from 'react';
import LoanEligibilitySDK from 'loan-eligibility-sdk';
import type { SDKEvent } from 'loan-eligibility-sdk/src/types';

type SDKStatus = 'idle' | 'loading' | 'success' | 'error' | 'completed';

interface UseLoanSDKOptions {
  phoneNumber: string;
  apiKey: string;
  apiSecret: string;
  partnerId: string;
  onResult?: (data: any) => void;
  onError?: (error: any) => void;
  autoRestore?: boolean;
}

export const useLoanSDK = (options: UseLoanSDKOptions) => {
  const [sdk, setSdk] = useState<LoanEligibilitySDK | null>(null);
  const [status, setStatus] = useState<SDKStatus>('idle');
  const [showSDK, setShowSDK] = useState(false);
  const restorationAttempted = useRef(false);

  useEffect(() => {
    if (options.phoneNumber.length !== 10) return;

    const sdkInstance = new LoanEligibilitySDK({
      apiKey: options.apiKey,
      apiSecret: options.apiSecret,
      phoneNumber: `+91${options.phoneNumber}`,
    });

    const handlers = {
      initiated: () => {
        setStatus('loading');
        setShowSDK(true);
      },
      ready: () => setStatus('success'),
      result: (event: SDKEvent) => {
        setStatus('completed');
        options.onResult?.(event.data);
      },
      error: (event: SDKEvent) => {
        setStatus('error');
        options.onError?.(event.data);
      },
      close: () => {
        if (status !== 'completed') {
          setStatus('idle');
          setShowSDK(false);
        }
      }
    };

    sdkInstance.on('INITIATED', handlers.initiated);
    sdkInstance.on('READY', handlers.ready);
    sdkInstance.on('ELIGIBILITY_RESULT', handlers.result);
    sdkInstance.on('ERROR', handlers.error);
    sdkInstance.on('CLOSE_FRAME', handlers.close);

    setSdk(sdkInstance);

    return () => {
      sdkInstance.off('INITIATED', handlers.initiated);
      sdkInstance.off('READY', handlers.ready);
      sdkInstance.off('ELIGIBILITY_RESULT', handlers.result);
      sdkInstance.off('ERROR', handlers.error);
      sdkInstance.off('CLOSE_FRAME', handlers.close);
    };
  }, [options.phoneNumber, options.apiKey, options.apiSecret]);

  const startEligibilityCheck = async () => {
    if (!sdk) throw new Error('SDK not initialized');

    try {
      await sdk.initialize({
        partnerId: options.partnerId,
        apiKey: options.apiKey,
        apiSecret: options.apiSecret,
      });

      sdk.openEligibilityCheck('inline');
    } catch (error) {
      setStatus('error');
      throw error;
    }
  };

  const closeSDK = () => {
    sdk?.closeFrame();
    setShowSDK(false);
    setStatus('idle');
  };

  return {
    sdk,
    status,
    showSDK,
    startEligibilityCheck,
    closeSDK,
  };
};

Constructor

Methods

Available methods for interacting with the SDK

initialize

async 
initialize(config: PartnerConfig): Promise<void>

Initializes the SDK with partner configuration and retrieves session data from the server.

openEligibilityCheck

sync 
openEligibilityCheck(mode: 'inline' | 'popup'): void

Opens the eligibility check form in either inline modal or popup window mode.

closeFrame

sync 
closeFrame(): void

Closes the SDK frame and cleans up all resources and event listeners.

on

sync 
on(event: string, handler: Function): void

Registers an event handler for SDK events like READY, ELIGIBILITY_RESULT, etc.

off

sync 
off(event: string, handler: Function): void

Removes a previously registered event handler for SDK events.

sendData

sync 
sendData(data: Record<string, unknown>): void

Sends custom data to the SDK iframe for processing or validation.

Events

Available events that the SDK emits during the loan eligibility process

INITIATED

lifecycle

Fired when the SDK frame is opened and initialization begins.

READY

lifecycle

Fired when the SDK is fully loaded and ready for user interaction.

ELIGIBILITY_RESULT

result

Fired when eligibility check is completed successfully with results.

ERROR

error

Fired when an error occurs during the eligibility check process.

CLOSE_FRAME

lifecycle

Fired when the SDK frame is closed by user action or system.

sdk.openEligibilityCheck(isMobile ? 'popup' : 'inline');

    sdk.openEligibilityCheck(isNowMobile ? 'popup' : 'inline');

});

üîÑ Error Handling

Comprehensive error handling with user feedback

Error Management 
sdk.on('ERROR', (event) => {
  const error = event.data.error;
  
  switch (error.code) {
    case 'NETWORK_ERROR':
      showErrorMessage('Network connection issue. Please check your internet connection.');
      break;
    case 'VALIDATION_ERROR':
      showErrorMessage(`Validation failed: ${error.message}`);
      break;
    case 'AUTH_ERROR':
      showErrorMessage('Authentication failed. Please contact support.');
      break;
    default:
      showErrorMessage('An unexpected error occurred. Please try again.');
  }
  
  // Log error for debugging
  console.error('SDK Error:', error);
  
  // Optional: Send to analytics
  analytics.track('sdk_error', {
    errorCode: error.code,
    errorMessage: error.message
  });
});

Troubleshooting

Common issues and solutions

‚ùå SDK not loading

Problem: The SDK frame appears blank or doesn't load.

Solution:

  • Check your API credentials are correct
  • Verify the API endpoint is accessible
  • Check browser console for JavaScript errors
  • Ensure your domain is whitelisted

üîê Authentication errors

Problem: Getting 401 or 403 errors during initialization.

Solution:

  • Verify your API key and secret are correct
  • Check if your partner account is active
  • Ensure you're using the correct environment (sandbox/production)

üì± Mobile responsiveness issues

Problem: SDK doesn't display properly on mobile devices.

Solution:

  • Use popup mode for mobile devices
  • Ensure viewport meta tag is set correctly
  • Test on different screen sizes

üîÑ State persistence issues

Problem: SDK state is lost on page refresh.

Solution:

  • Implement localStorage persistence as shown in advanced examples
  • Handle restoration logic properly
  • Clear state when flow is completed
